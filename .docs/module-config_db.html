Gets a database client for executing transactions

Description:

Source:

Example
// Using a transaction
const client = await getClient();
try {
  await client.query('BEGIN');
  // ... perform multiple queries
  await client.query('COMMIT');
} catch (e) {
  await client.query('ROLLBACK');
  throw e;
} finally {
  client.release();
}
