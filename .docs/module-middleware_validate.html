Example
// Define validation schema for login
const loginSchema = {
  body: z.object({
    email: z.string().email("Invalid email format"),
    password: z.string().min(8, "Password must be at least 8 characters")
  })
};

// Use in routes
router.post('/login', validate(loginSchema), authController.login);
